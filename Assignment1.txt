import java.util.Arrays;
import java.util.Scanner;

public class TodoManager{
    
    public String[] task(){
        Scanner scanner = new Scanner(System.in);
        System.out.println("Please type your 5 task for today, each task should end with a period:");
        String input = scanner.nextLine();
        return input.split("\\.");
    }
    
    public void IncreasingOrder(String[] tasks) {
        Arrays.sort(tasks, (a, b) -> Character.compare(a.charAt(0), b.charAt(0)));
        System.out.println("Tasks in increasing order:");
        for (String task : tasks) {
            System.out.println(task);
        }
    }
    
    public void DecreasingOrder(String[] tasks) {
        Arrays.sort(tasks, (a, b) -> Character.compare(b.charAt(0), a.charAt(0)));
        System.out.println("Tasks in decreasing order:");
        for (String task : tasks) {
            System.out.println(task);
        }
    }
    
    public String checkRepeatedTasks(String[] tasks) {
        for (int i = 0; i < tasks.length; i++) {
            for (int j = i + 1; j < tasks.length; j++) {
                if (tasks[i].equals(tasks[j]) && !tasks[i].isEmpty()) {
                    return tasks[i];
                }
            }
        }
        return null;
    }
    
	public static void main(String[] args){
	    Scanner scanner = new Scanner(System.in);
        System.out.println("Hi, Please enter your name:");
        String name = scanner.nextLine();
        System.out.println("Hi "+ name);
        
        TodoManager todoManager = new TodoManager();
        String[] tasks = todoManager.task();
        System.out.println("Would you like to see all the tasks in increasing or decreasing order? Press 1 for increasing order or 2 for decreasing order.");
        int order = scanner.nextInt();
        scanner.nextLine();
        
        if (order == 1) {
            todoManager.IncreasingOrder(tasks);
        } 
        else{
            todoManager.DecreasingOrder(tasks);
        }
        
        String repeatedTask = todoManager.checkRepeatedTasks(tasks);
        if (repeatedTask != null) {
            System.out.println("Repeated task found: " + repeatedTask);
        } 
        else {
            System.out.println("No repeated tasks found.");
        }
    }

}
